135 	private void tablaMaterias(CursoPropio curso) {
136 		materiasCurso.addColumn("Nombre");
137 		materiasCurso.addColumn("Responsable");
138 		materiasCurso.addColumn("Horas");
139 		materiasCurso.addColumn("Fecha Inicio");
140 		materiasCurso.addColumn("Fecha Fin");
141 		
142 		Collection<Materia> materias = curso.materias;
143 		if (materias!=null) {
144 			Iterator<Materia> ite = materias.iterator();
145 			while(ite.hasNext()){
146 				Materia temp = ite.next();
147 				materiasCurso.addRow(new Object[] { temp.getNombre(), temp.responsable, temp.getHoras(), temp.getFechaInicio() ,temp.getFechaFin() });
148 			}
149 		}
150 
151 
152 		JTable materiasTable = new JTable(materiasCurso){
153 			@Override
154 			public boolean isCellEditable(int rowIndex, int colIndex) {
155 				return false; //Disallow the editing of any cell
156 			}
157 		};
158 
159 		materiasTable.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
160 
161 		JScrollPane scrollLista = new JScrollPane(materiasTable);
162 		scrollLista.setBounds(306, 51, 400, 188);
163 		getContentPane().add(scrollLista);
164 	}

166 	private void botonesLayout(final int type, final ProfesorUCLM director, final CursoPropio curso) {
167 		JButton btnNewButton = new JButton("Atras");
168 		btnNewButton.addActionListener(new ActionListener() {
169 			public void actionPerformed(ActionEvent e) {
170 				if (type == 0 || type == 1) new PantallaCursosAprobados(type, director);
171 				else new PantallaEmpleadosVicerrectorado();
172 				setVisible(false);
173 			}
174 		});
175 		btnNewButton.setBounds(10, 513, 200,30);
176 		getContentPane().add(btnNewButton);
177 
178 		JButton btnMatricularEvaluar = new JButton("Matricular");
179 		btnMatricularEvaluar.addActionListener(new ActionListener() {
180 			public void actionPerformed(ActionEvent e) {
181 				
182 				int confirm = 1;
183 
184 				if (type == 0) { 
185 					confirm = JOptionPane.showConfirmDialog(null,"¿Matricular?","Matricular en curso",JOptionPane.YES_NO_OPTION, 1);
186 				} else { 
187 					confirm = JOptionPane.showConfirmDialog(null,"¿Evaluar propuesta?","Evaluar propuesta",JOptionPane.YES_NO_OPTION, 1);
188 				}
189 				
190 				if(confirm == 0)  {
191 					if (type == 0) { // TODO -- matricular
192 						new PantallaMatriculacion(director, curso); //aqui hay que meter alumno
193 		                setVisible(false);
194 					}	
195 					else {  // Evaluar
196 						
197 						if (!evaluarPropuestaCorrecto(type)) return;
198 						
199 						GestorPropuestasCursos gestor = new GestorPropuestasCursos();
200 						if (aceptarPropuesta.isSelected()) curso.estado = EstadoCurso.VALIDADO;
201 						else curso.estado = EstadoCurso.PROPUESTA_RECHAZADA;
202 						
203 						try {
204 							gestor.editarPropuestaCurso(curso);
205 						} catch (Exception e1) {
206 							e1.printStackTrace();
207 							
208 						}
209 						
210 						new PantallaEmpleadosVicerrectorado();
211 						setVisible(false);
212 					}
213 					
214 				}
215 			}
216 		});
217 		btnMatricularEvaluar.setBounds(574, 513, 200,30);
218 		getContentPane().add(btnMatricularEvaluar);
219 
220 		// CAMBIOS DE BOTONES
221 		if (type == 1) btnMatricularEvaluar.setVisible(false);
222 		else if (type == 2) btnMatricularEvaluar.setText("Evaluar");		
223 	}

225 	public boolean evaluarPropuestaCorrecto(final int type) {
226 		boolean result = true;
227 		if (!aceptarPropuesta.isSelected() && !rechazarPropuesta.isSelected()) {
228 			aceptarPropuesta.setBackground(new Color(222, 129, 122));
229 			rechazarPropuesta.setBackground(new Color(222, 129, 122));
230 			result = false;
231 		}else {
232 			aceptarPropuesta.setBackground(getBackground());
233 			rechazarPropuesta.setBackground(getBackground());
234 		}
235 		
236 		if (type == 2 && rechazarPropuesta.isSelected()) {
237 			if (descripcion.getText().equals("")) {
238 				descripcion.setBackground(new Color(222, 129, 122));
239 				result = false;
240 			} else {
241 				descripcion.setBackground(new Color(255, 255, 255));
242 			}
243 		}
244 		
245 		return result;
246 	}
